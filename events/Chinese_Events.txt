#战国开始
country_event = { 
	id = ShiJian.1
    title = ShiJian.1.t
	desc = ShiJian.1.d
	picture = military_parade
	major = yes
	trigger = { 
		OR = { 
			tag = ZHO
			AND = { 
				ai = no
				NOT = { exists = ZHO }
			}
		}
		not = { 
			has_global_flag = zhanguokaishi
		}
		OR = { 
			AND = { 
				is_year = 274
				not = { 
					hre_reform_level = 4
				}
			}
			AND = { 
				is_year = 306
				not = { 
					hre_reform_level = 5
				}
			}
			AND = { 
				is_year = 336
				not = { 
					hre_reform_level = 6
				}
			}
			AND = { 
				is_year = 376
				not = { 
					hre_reform_level = 7
				}
			}
		}
	}
	mean_time_to_happen = { 
		months = 48
	}
	
	option = { 
		name = "ShiJian.1.a"
		set_global_flag = zhanguokaishi
	}
}

#弭兵之盟
country_event = { 
	id = ShiJian.9
	title = ShiJian.9.t
	picture = military_parade
	desc = ShiJian.9.d	
	
	fire_only_once = yes
	
	major = yes
	
	trigger = { 
	    is_emperor = yes
		is_at_war = no
		NOT = {
			any_known_country = {
				is_in_league_war = yes
			}
		}		
		has_regency = no	
		hre_religion_locked = no
		hre_religion_treaty = no
		OR = {
			AND = {
				hre_leagues_enabled = yes
				had_global_flag = {
					flag = evangelical_union_happened
					days = 18250
				}
                religion = chinese_religion				
				NOT = {
					any_known_country = {
						is_elector = yes
						religion = liyuebadao
						is_league_enemy = ROOT
						truce_with = ROOT
					}		
				}
			}
			AND = {
				hre_leagues_enabled = yes
				had_global_flag = {
					flag = evangelical_union_happened
					days = 18250
				}
                religion = liyuebadao				
				NOT = {
					any_known_country = {
						is_elector = yes
						religion = chine
						is_league_enemy = ROOT
						truce_with = ROOT
					}		
				}
			}
		}
	}
	
	mean_time_to_happen = { 
		months = 300
	}
	
	option = { #接纳
		name = "ShiJian.9.a"
		set_hre_religion_treaty = yes
		add_imperial_influence = -10
		every_country = {
		  limit = {
		  is_part_of_hre = yes
		  NOT = { religion = chinese_religion }
		  }
		  add_opinion = { who = ROOT modifier = opinion_JieNaManYi }
		}
	}
	
	option = { #接纳
		name = "ShiJian.9.b"
		add_imperial_influence = 10
		set_hre_religion_locked = yes
		every_country = {
		  limit = {
		  is_part_of_hre = yes
		  NOT = { religion = chinese_religion }
		  }
		  add_opinion = { who = ROOT modifier = opinion_JieNaManYi_Wei }
		}
	}
	
}
