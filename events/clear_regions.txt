#Clear Britannia

country_event = {
	id = clear_regions.201
	title = "clear_regions.201.t"
	desc = "clear_regions.201.d"
	picture = senat
	
	is_triggered_only = yes
	
	trigger = {
		
	}

	option = { #Colonizable disabling
		name = "clear_regions.201.a"
		hidden_effect = {
			britannia_superregion = {
				limit = {
					NOT = { owned_by = ROOT }
				}
				cede_province = XXX
				add_base_manpower = -999
				add_base_production = -999
				add_base_tax = -999
				remove_all_buildings_scripted_effect = yes
			}
		}
	}

	option = { #Complete disabling
		name = "clear_regions.201.b"
		hidden_effect = {
			set_global_flag = britannia_disabled_flag
			britannia_superregion = {
				cede_province = XXX
				add_base_manpower = -999
				add_base_production = -999
				add_base_tax = -999
				remove_all_buildings_scripted_effect = yes
			}
		}
	}
}
