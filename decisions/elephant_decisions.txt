country_decisions = {
	
	import_indian_elephants_decision = {
		potential = {
			NOT = { capital_scope = { continent = india } }
			NOT = { capital_scope = { continent = china } }
			NOT = { ivory = 1 }
			any_known_country = {
				OR = {
					capital_scope = { continent = india }
					capital_scope = { continent = china }
				}
				ivory = 1
				has_institution = elephant_domestication
				border_distance = {
					who = ROOT
					distance = 150
				}
			}
			NOT = { num_of_cossacks = 1 }
			neither_tribal_nor_native_trigger = yes
		}
		allow = {
			NOT = {
				OR = {
					has_institution = elephant_domestication
					ivory = 1
				}
			}
			treasury = 250
		}
		effect = {
			random_known_country = {
				limit = {
					capital_scope = { continent = india }
					ivory = 1
					has_institution = elephant_domestication
				}
				country_event = { 
					id = elephants.7
					days = 7
				}
			}
			hidden_effect = {
				set_country_flag = imported_elephants_flag
			}
		}
		ai_importance = 1	
		ai_will_do = {
			factor = 1
			modifier = {
				factor = 0
				NOT = { treasury = 250 }
			}
			modifier = {
				factor = 1.25
				treasury = 500
			}
		}
	}
	
}
