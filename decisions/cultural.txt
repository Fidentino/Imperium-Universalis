country_decisions = {
	########################################
	#
	#		Hold a Festival
	#
	########################################
	
	 hold_a_festival = {
		potential = {
			NOT = { is_tribal = yes }
			NOT = { government_rank = 2 }
		}
		allow = {
			treasury = 100
			dip_power = 20
			absolutism = 2 
			capital_scope = { NOT = { has_province_modifier = festival_hold_modifier } }
		}
		effect = {
			add_treasury = -100
			add_dip_power = -20
			add_absolutism = -2 
			add_splendor = 25
			add_legitimacy = 5
			add_republican_tradition = 3
			add_devotion = 3
			capital_scope = {
				area = {
					limit = {
						owned_by = ROOT
					}
					add_province_modifier = {
						name = festival_hold_modifier
						duration = 365
					}
				}
			}
			if = {
				limit = {
					has_country_modifier = polis_tyrannis_modifier
				}
				add_faction_influence = {
					faction = fac_monarchists
					influence = 5
				}
			}
		}
		ai_will_do = {
			factor = 1
			modifier = {
				factor = 0
					NOT = { years_of_income = 0.25 }
			}
			modifier = {
				factor = 0
					NOT = { dip_power = 100 }
			}
		}
	}
}
