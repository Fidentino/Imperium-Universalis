estate_nobility =
{
	icon = 7

	# If true, country will get estate
	trigger = {
		OR = {
			always = yes
			has_government_attribute = enables_nobility
		}
	}

	# These scale with loyalty & power
	country_modifier_happy = {
		global_tax_modifier = 0.2
		cavalry_power = 0.10
	}
	country_modifier_neutral = {
		global_tax_modifier = 0.2
	}	
	country_modifier_angry = {
		global_tax_modifier = -0.2
		global_unrest = 2
	}

	land_ownership_modifier = {
		nobility_loyalty_modifier = 0.2
	}

	province_independence_weight = {
		factor = 1
		modifier = {
			factor = 0.1
			NOT = { religion_group = owner }
		}
		modifier = {
			factor = 0.5
			NOT = { religion = owner }
		}
		modifier = {
			factor = 0.67
			NOT = { is_state_core = owner }
		}
		modifier = {
			factor = 1.2
			base_tax = 10
		}
	}

	# Influence modifiers
	base_influence = 10
	
	custom_name = {
		desc = estate_aristocrats
		trigger = {
			OR = {
				hass_government_reform = oligarchy_reform
			}
		}
	}

	color = { 200 150 0 }

	privileges = {
		estate_nobility_captains
		estate_nobility_generals
		estate_nobility_estates
		estate_nobility_governorship
		estate_nobility_advisors
		estate_nobility_slaves
	}

	agendas = {
		estate_nobility_develop_x
		estate_nobility_hire_advisor
		estate_nobility_fire_advisor
		estate_nobility_subjugate_neighbour
		estate_nobility_increase_alliances
	}
	influence_from_dev_modifier = 1.0
}
