
#Ask for Socii
ask_for_socii_subject = {
	category = influence
	alert_index = 30
	alert_tooltip = ask_for_socii_subject_alert_tooltip.1
	
	require_acceptance = yes
	is_visible = {
		is_subject = no
		OR = {
			has_reform = res_publica_reform
			has_reform = roman_kingdom_reform
		}
		government_rank = 2
		FROM = {
			capital_scope = {
				OR = {
					region = italian_region
					area = lucania_area
					area = apulia_area
					area = salento_area
					area = samnium_area
				}
			}
			is_subject = no
		}
	}
	is_allowed = {
		hidden_trigger = {
			is_subject = no
			OR = {
				has_reform = res_publica_reform
				has_reform = roman_kingdom_reform
			}
			government_rank = 2
			FROM = {
				capital_scope = {
					OR = {
						region = italian_region
						area = lucania_area
						area = apulia_area
						area = salento_area
						area = samnium_area
					}
			}
			is_subject = no
			}
		}
		alliance_with = FROM
		reverse_has_opinion = {
			who = FROM
			value = 150
		}
		FROM = {
			is_at_war = no
			NOT = { war_with = ROOT }
			NOT = { has_country_modifier = recently_removed_socii }
		}
		is_bankrupt = no
		NOT = { war_with = FROM }
	}
	on_accept = {
		add_prestige = 2
		create_subject = {
			subject_type = socii_subject
			subject = FROM
		}
	}
	
	ai_acceptance = {
		add_entry = {
			name = TOTAL_DEVELOPMENT
			export_to_variable = {
				variable_name = ai_value
				value = total_development
				who = FROM
			}
			multiply_variable = { which = ai_value value = -2 }
		}
		add_entry = {
			name = TRUST
			export_to_variable = {
				variable_name = ai_value
				value = trust
				who = FROM
				with = ROOT
			}
			subtract_variable = { which = ai_value value = 70 }
		}
		add_entry = {
			name = ARMY_SIZE
			export_to_variable = {
				variable_name = ai_value
				value = land_forcelimit
				who = ROOT
			}
			change_variable = { which = ai_value value = 1 }
		}
		add_entry = {
			name = STABILITY
			export_to_variable = {
				variable_name = ai_value
				value = stability
				who = ROOT
			}
			multiply_variable = { which = ai_value value = 5 }
		}
		add_entry = {
			name = PRESTIGE
			export_to_variable = {
				variable_name = ai_value
				value = prestige
				who = ROOT
			}
			divide_variable = { which = ai_value value = 10 }
		}
		add_entry = {
			name = DIPLOMATIC_REPUTATION
			export_to_variable = {
				variable_name = ai_value
				value = modifier:diplomatic_reputation
				with = ROOT
			}
			multiply_variable = { which = ai_value value = 6 }
		}
		add_entry = {
			name = DIFFERENT_RELIGION
			limit = { NOT = { religion = FROM } }
			export_to_variable = {
				which = ai_value
				value = -10
			}
		}
		add_entry = {
			name = DIFFERENT_CULTURE_GROUP
			limit = { NOT = { culture_group = FROM } }
			export_to_variable = {
				which = ai_value
				value = -10
			}
		}
		add_entry = {
			name = BASE_VALUE_CUSTOM
			limit = { FROM = { ai = yes } }
			export_to_variable = {
				which = ai_value
				value = -15
			}
		}
	}
	
	ai_will_do = {
		has_any_disaster = no
		FROM = {
			is_neighbor_of = ROOT
			NOT = { army_size = ROOT }
			NOT = { total_development = ROOT }
		}
		reverse_has_opinion = {
			who = FROM
			value = 150
		}
		stability = 1
		army_size = 10
		is_at_war = no
		NOT = { average_unrest = 1 }
	}
}

#Frees Socii
remove_socii_subject = {
	category = influence
	alert_index = 41
	alert_tooltip = remove_socii_subject_alert_tooltip.1
	
	require_acceptance = no
	is_visible = {
		FROM = {
			is_subject_of_type = socii_subject
			is_subject_of = ROOT
		}
	}
	is_allowed = {
		hidden_trigger = {
			FROM = {
				is_subject_of_type = socii_subject
				is_subject_of = ROOT
			}
		}
		is_at_war = no
		has_any_disaster = no
		FROM = {
			is_subject_of_type = socii_subject
			is_subject_of = ROOT
		}
	}
	on_accept = {
		FROM = {
			grant_independence = yes
			add_truce_with = ROOT
			add_country_modifier = {
				name = recently_removed_socii
				duration = 7300
			}
		}
		add_prestige = -5
		add_mil_power = -20
	}
	
	ai_will_do = {
		FROM = {
			OR = {
				total_development = ROOT
				liberty_desire = 80
			}
		}
	}
}
