
bazhu_end_war =  {
	category = influence
	require_acceptance = yes
	alert_index = 2
	alert_tooltip = ICON_PEACE
	is_visible = {
		has_country_flag = china_bazhu
		FROM = { religion = ROOT }
	}
	is_allowed = {
		any_known_country = {
		   religion = FROM
		   defensive_war_with = FROM
		   NOT = { tag = ROOT }
		   NOT = { war_with = ROOT }
		}
		is_subject = no
		prestige = 10
	}
	on_accept = {
		every_known_country = {
		   limit = {
		      religion = FROM
		      defensive_war_with = FROM
		      NOT = { tag = ROOT }
		   }
		   white_peace = FROM
		   country_event = { id = ShiJian.25 }
		}
		FROM = {
		   country_event = { id = ShiJian.25 }
		}
	}
	on_decline = {
	    FROM = {
		  set_ruler_flag = refuse_end_war
		}
		declare_war_with_cb = {
		  who = FROM
		  casus_belli = cb_insult
        }
		add_prestige = -10
	}
	ai_will_do = {
		is_at_war = no
	}
}
