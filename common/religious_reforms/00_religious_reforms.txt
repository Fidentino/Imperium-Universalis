pagan_reforms_superior_religion = {
	trigger = {
		unreformed_pagan_religious_group_trigger = yes
		has_country_flag = proselytism_religion_reforms_path_flag
	}	

	can_buy_idea = {
		NOT = {
			government = native
			government = tribal
		}
		OR = {
			has_institution = writing_system
			has_institution = bureaucracy_institution
		}
		legitimacy_equivalent = 100
		is_at_war = no
		has_any_disaster = no
		NOT = { num_of_revolts = 1 }
		stability = 3
		if = {
			limit = {
				uses_patriarch_authority = yes
			}
			patriarch_authority = 1
		}
		else = { absolutism = 20 }
		adm_power = 200
		dip_power = 200
		mil_power = 200
		num_of_cities = 20
	}
	
	pagan_reforms_superior_religion_1 = {
		prestige_per_development_from_conversion = 0.5
	}
	pagan_reforms_superior_religion_2 = {
		missionary_maintenance_cost = -0.1
	}
	pagan_reforms_superior_religion_3 = {
		global_heathen_missionary_strength = 0.02
	}
	pagan_reforms_superior_religion_4 = {
		global_heretic_missionary_strength = 0.02
	}
	pagan_reforms_superior_religion_5 = {
		missionaries = 1
	}	
	
	ai_will_do = {
		factor = 1
	}
}
