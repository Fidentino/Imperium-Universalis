slaves_find_coordinates = {
	
	# X, total slave population %
	if = { 		limit = { check_variable = { slave_pop_social = 200 } } set_country_flag = slaves_x7 }
	else_if = { limit = { check_variable = { slave_pop_social = 120 } } set_country_flag = slaves_x6 }
	else_if = { limit = { check_variable = { slave_pop_social = 80 } }  set_country_flag = slaves_x5 }
	else_if = { limit = { check_variable = { slave_pop_social = 50 } }  set_country_flag = slaves_x4 }
	else_if = { limit = { check_variable = { slave_pop_social = 25 } }  set_country_flag = slaves_x3 }
	else_if = { limit = { check_variable = { slave_pop_social = 10 } }  set_country_flag = slaves_x2 }
	else_if = { limit = { check_variable = { slave_pop_social = 0 } }   set_country_flag = slaves_x1 }
		
	# Y, slave population % adjusted to productivity
	
	if = { 		limit = { check_variable = { slave_pop_economic = 300 } } set_country_flag = slaves_y15 }
	else_if = { limit = { check_variable = { slave_pop_economic = 250 } } set_country_flag = slaves_y14 }
	else_if = { limit = { check_variable = { slave_pop_economic = 200 } } set_country_flag = slaves_y13 }
	else_if = { limit = { check_variable = { slave_pop_economic = 160 } } set_country_flag = slaves_y12 }
	else_if = { limit = { check_variable = { slave_pop_economic = 140 } } set_country_flag = slaves_y11 }
	else_if = { limit = { check_variable = { slave_pop_economic = 120 } } set_country_flag = slaves_y10 }
	else_if = { limit = { check_variable = { slave_pop_economic = 100 } } set_country_flag = slaves_y9 }
	else_if = { limit = { check_variable = { slave_pop_economic = 80 } }  set_country_flag = slaves_y8 }
	else_if = { limit = { check_variable = { slave_pop_economic = 60 } }  set_country_flag = slaves_y7 }
	else_if = { limit = { check_variable = { slave_pop_economic = 50 } }  set_country_flag = slaves_y6 }
	else_if = { limit = { check_variable = { slave_pop_economic = 40 } }  set_country_flag = slaves_y5 }
	else_if = { limit = { check_variable = { slave_pop_economic = 30 } }  set_country_flag = slaves_y4 }
	else_if = { limit = { check_variable = { slave_pop_economic = 20 } }  set_country_flag = slaves_y3 }
	else_if = { limit = { check_variable = { slave_pop_economic = 10 } }  set_country_flag = slaves_y2 }
	else_if = { limit = { check_variable = { slave_pop_economic = 0 } }   set_country_flag = slaves_y1 }
	
	
	
}
