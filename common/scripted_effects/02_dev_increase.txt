
increase_manpower_dev_effect = {
	if = {
		limit = {
			OR = {
				owner = {
					has_global_modifier_value = {
						which = development_cost
						value = 0.0
					}
				}
				has_local_modifier_value = {
					which = local_development_cost
					value = 0.0
				}
			}
			NOT = {
				owner = {
					has_global_modifier_value = {
						which = development_cost
						value = 0.1
					}
					has_global_modifier_value = {
						which = development_cost
						value = -0.1
					}
				}
				has_local_modifier_value = {
					which = local_development_cost
					value = 0.1
				}
				
				has_local_modifier_value = {
					which = local_development_cost
					value = -0.1
				}
			}
		}
		owner = { add_mil_power = -200 }
		add_base_manpower = 1
	}
}
